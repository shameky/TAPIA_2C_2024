106static void turnon_LEDs_distance(void *pvParameter)
107{
108
109 while(true)
110 {
111 if (MEASURE_DISTANCE)
112 {
113 if (distance < first_measurement_value)
114 {
115 LedOff(LED_1);
116 LedOff(LED_2);
117 LedOff(LED_3);
118 }
119
120 else if (first_measurement_value <= distance && distance < second_measurement_value)
121 {
122 LedOn(LED_1);
123 LedOff(LED_2);
124 LedOff(LED_3);
125 }
126
127 else if(second_measurement_value <= distance && distance < third_measurement_value)
128 {
129 LedOn(LED_1);
130 LedOn(LED_2);
131 LedOff(LED_3);
132 }
133
134 else
135 {
136 LedOn(LED_1);
137 LedOn(LED_2);
138 LedOn(LED_3);
139 }
140 }
141
142 else
143 {
144 LedOff(LED_1);
145 LedOff(LED_2);
146 LedOff(LED_3);
147 }
148 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
149 }
