93static void turnon_LEDs_distance(void *pvParameter)
94{
95
96 while(true)
97 {
98 if (MEASURE_DISTANCE)
99 {

101 {
102 LedOff(LED_1);
103 LedOff(LED_2);
104 LedOff(LED_3);
105 }
106
107 else if (first_measurement_value <= distance && distance < second_measurement_value)
108 {
109 LedOn(LED_1);
110 LedOff(LED_2);
111 LedOff(LED_3);
112 }
113
114 else if(second_measurement_value <= distance && distance < third_measurement_value)
115 {
116 LedOn(LED_1);
117 LedOn(LED_2);
118 LedOff(LED_3);
119 }
120

122 {
123 LedOn(LED_1);

169void static read_switches(void *pvParameter)
170{
171 while(true)
172 {
173 uint8_t keys = SwitchesRead();
174 switch (keys)
175 {
176 case SWITCH_1:
177 MEASURE_DISTANCE = !MEASURE_DISTANCE;
178 break;
179
180 case SWITCH_2:
181 hold_measurement = !hold_measurement;
182 break;
183 }
184 vTaskDelay(CONFIG_SWITCH_PERIOD / portTICK_PERIOD_MS);
